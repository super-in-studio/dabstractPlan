public class test {

public static void main(String[] args){

//必须有序

int[] arrs={1,2,3,4,5,6} ;

int find=5;

int result = binarySearch(arrs,find) ;

System.out.println("二分查找"+result) ;

}


public static int binarySearch(int[] arrs,int find){

int left = 0 ;

int right = arrs.length -1 ;

int min ;

while(left < right){

min = (left + right)/2

if(arrs[min] > find){

right -- ;

}else if(arrs[min] < find){

left++ ;

}else{

return min

}


return -1 ;

}


}

}
