public static void main(String[] args) {

    

    int[] arrs={6,5,4,3,2,1} ;

    sortArrs(arrs) ;

    for(int i=0;i<arrs.length ;i++){

     System.out.println(arrs[i]);

    }

  }

   

  private static void sortArrs(int[] arrs){

   for(int i=1;i<arrs.length ; i++){

   int j = i-1 ;

   int value = arrs[i] ;

   for(; j>=0 ; j--){

   if(value < arrs[j]){

   arrs[j+1] = arrs[j] ;

   }

   }

   arrs[j+1] = value ;

   }

  }
