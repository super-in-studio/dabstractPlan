public static void main(String[] args) {

   int[] arrs = {6,7,8,1,2,3,4} ;

   sortArrs(arrs) ;

   for(int i =0 ; i < arrs.length ; i++){

   System.out.println(arrs[i]);

   }

  }

   

  private static void sortArrs(int[] arrs){

   for(int i=0 ; i<arrs.length ; i++){

   for(int j=i ; j<arrs.length ; j++){

   if(arrs[i] > arrs[j]){

   arrs[j] ^= arrs[i] ;

   arrs[i] ^=arrs[j] ;

   arrs[j] ^=arrs[i] ;

   }

   }

   }

  }
