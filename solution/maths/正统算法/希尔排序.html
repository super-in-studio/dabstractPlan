public static void main(String[] args) {

   int[] arrs = {9,8,7,6,5,4,3,2,1} ;

   int gap = arrs.length/2 ;

  

   while(gap > 0){

   //定义列排序次数

   for(int i=0 ; i<gap ; i++){

   //类冒泡算法

   for(int j=0 ; j<arrs.length ; j=j+gap){

    for(int k=j+gap ; k<arrs.length ; k=k+gap){

    if(arrs[k] < arrs[j]){

   arrs[k] ^= arrs[j] ;

   arrs[j] ^= arrs[k] ;

   arrs[k] ^= arrs[j] ;

    }

    }

   }

  

   }

   gap = gap/2 ; 

   }

  

  

   //打印结果

   for(int i=0 ; i<arrs.length ; i++){

   System.out.println(arrs[i]) ;

   }

  }
