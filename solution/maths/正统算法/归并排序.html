public static void main(String[] args) {

   int[] arrs = {6,5,4,3,2,1,9,4} ;

   sort(arrs,0,arrs.length-1) ;

   for(int i=0;i<arrs.length ; i++){

   System.out.println(arrs[i]) ;

   }

  }

   

  private static void sort(int[] arrs,int left,int right){

   int mid = (left + right)/2 ;

   if(right > left){

   sort(arrs,left,mid) ;

   sort(arrs,mid+1,right) ;

   merge(arrs,left,mid,right) ;

   }

  }

   

  private static void merge(int[] arrs,int left ,int mid , int right){

   int[] temp = new int[arrs.length] ;

   int p1=left , p2=mid+1 , k=left ;

   while(p2<=right && p1 <= mid){

   if(arrs[p1]<=arrs[p2]){

   temp[k++] = arrs[p1++] ;

   }else{

   temp[k++] = arrs[p2++] ;

   }

   }

    

   while(p1<=mid){

   temp[k++]=arrs[p1++] ;

   }

    

   while(p2<right){

   temp[k++] = arrs[p2++] ;

   }

    

   for(int i=left ;i<k ;i++){

   arrs[i] = temp[i];

   }

  }
