package com.lxtx.hello.wallet.web.config;


import org.springframework.context.annotation.Bean;

import org.springframework.context.annotation.Configuration;

import org.springframework.context.annotation.DependsOn;


import com.lxtx.hello.common.interceptor.LocaleInterceptor;

import com.lxtx.hello.common.log.interceptor.LoggerInterceptor;

import com.lxtx.hello.common.manager.RedisLockManager;

import com.lxtx.hello.common.security.interceptor.CheckSignatureInterceptor;

import com.lxtx.hello.common.security.interceptor.FeignRequestInterceptor;

import com.lxtx.hello.common.utils.RedisCacheUtils;

import com.lxtx.hello.common.utils.SmsUtil;

import com.lxtx.hello.common.utils.environment.PropertiesUtil;

import com.lxtx.hello.common.utils.event.EvenPublishUtils;

import com.lxtx.hello.common.utils.exchange.rate.niu.NiuChaExchangeRateUtils;

import com.lxtx.hello.common.utils.sixx.SixxClient;


/**

 * @author zhiHua chen Created data on 2018/08/07

 */

@Configuration

public class Beans {


@Bean("propertiesUtil")

public PropertiesUtil propertiesUtil() {

return new PropertiesUtil();

}


@Bean

@DependsOn("propertiesUtil")

public LoggerInterceptor loggerInterceptor() {

return new LoggerInterceptor();

}


@Bean

@DependsOn("propertiesUtil")

public CheckSignatureInterceptor checkSignatureInterceptor() {

return new CheckSignatureInterceptor();

}


@Bean

public LocaleInterceptor localeInterceptor() {

return new LocaleInterceptor();

}


@Bean

@DependsOn("propertiesUtil")

public FeignRequestInterceptor feignRequestInterceptor() {

return new FeignRequestInterceptor();

}


@Bean

public RedisCacheUtils redisCacheUtils() {

return new RedisCacheUtils();

}


@Bean

public SixxClient sixxClient() {

return new SixxClient();

}


@Bean

public NiuChaExchangeRateUtils niuChaExchangeRateUtils() {

return new NiuChaExchangeRateUtils();

}


/**

* 短信工具初始 化

* 

* @return

*/

@Bean

@DependsOn("propertiesUtil")

public SmsUtil getSmsUtil() {

SmsUtil smsUtil = new SmsUtil();

smsUtil.setUsername(PropertiesUtil.getString(PropertiesUtil.ISMS_USERNAME));

smsUtil.setPasseordIsm(PropertiesUtil.getString(PropertiesUtil.ISMS_PASSEORDISM));

smsUtil.setBaseurl(PropertiesUtil.getString(PropertiesUtil.ISMS_BASEURL));

//国外短信

smsUtil.setAccount(PropertiesUtil.getString(PropertiesUtil.YUNSMS_ACCOUNT));

smsUtil.setPassword(PropertiesUtil.getString(PropertiesUtil.YUNSMS_PASSWORD));

smsUtil.setUrl(PropertiesUtil.getString(PropertiesUtil.YUNSMS_URL));

return smsUtil;

}

/**

* 初始化spring event 发送工具

* @param applicationEventPublisher

* @return

*/

@Bean

public EvenPublishUtils evenPublishUtils(){

return EvenPublishUtils.getInstance();

}

@Bean

public RedisLockManager redisLockManager(){

return new RedisLockManager();

}


}
